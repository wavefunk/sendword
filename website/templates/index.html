— 03 / CONFIG

One file. Full control.

Everything lives in sendword.toml. Export as JSON, import on another machine, version-control the whole thing.

# sendword.toml
[[hooks]]
name = "Deploy"
slug = "deploy"

[hooks.executor]
type = "shell"
command = "bash deploy.sh { branch }"

[hooks.auth]
mode = "bearer"
token = "${DEPLOY_TOKEN}"

[hooks.trigger_rules]
cooldown = "5m"

[[hooks.trigger_rules.payload_filters]]
field = "branch"
operator = "regex"
value = "^(main|release/.*)"
