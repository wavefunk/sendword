How it works

01

Configure a hook

[[hooks]]
name = "Deploy"
slug = "deploy"

[hooks.executor]
type = "shell"
command = "bash deploy.sh { branch }"

[hooks.auth]
mode = "bearer"
token = "${DEPLOY_TOKEN}"
02

Add rules & guards

[hooks.trigger_rules]
cooldown = "5m"

[[hooks.trigger_rules.payload_filters]]
field = "branch"
operator = "regex"
value = "^(main|release/.*)"

[hooks.concurrency]
mode = "queue"
queue_depth = 5
03

Run sendword

$ sendword serve
listening on 127.0.0.1:8080

# hooks available:
#   POST /hook/deploy
#   POST /hook/notify
#   POST /hook/sync

# dashboard at /

One file. Full control.

Everything lives in sendword.toml. Export as JSON, import it on another machine, version-control the whole thing.

[server]
bind = "0.0.0.0"
port = 8080

[defaults.timeout]
timeout = "30s"

[defaults.retries]
count = 3
backoff = "exponential"
initial_delay = "2s"
max_delay = "60s"

[masking]
env_vars = ["API_KEY", "DB_PASSWORD"]
patterns = ["(token)[=:]\\s*\\S+"]

[backup]
bucket = "my-backups"
schedule = "0 2 * * *"
retention.max_count = 30
cargo install sendword
